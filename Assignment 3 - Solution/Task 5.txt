#include <iostream>

using namespace std;


int main()
{
	char array[22] = "StopAndSmellTheRoses.";

	for (int i = 0; i < 22; i++)
	{
		if (i > 1 && array[i] >= 65 && array[i] <= 90)
		{
			cout << " ";
			cout << char((int(array[i]) + 32));
		}
		else
		{
			cout << array[i];
		}
	}

	return 0;
}
